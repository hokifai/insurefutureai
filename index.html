<script>
   // --- Modal Functionality ---

   const demoModal = document.getElementById('demoModal');
   const modalMessage = document.getElementById('modalMessage');

   /**
    * Opens the demo request modal.
    */
   function openModal() {
       demoModal.classList.remove('hidden');
       modalMessage.classList.add('hidden'); // Hide thank you message on open
   }

   /**
    * Closes the demo request modal.
    */
   function closeModal() {
       demoModal.classList.add('hidden');
   }

   /**
    * Handles the form submission. Logs data and shows a thank you message.
    * @param {Event} event - The form submission event.
    */
   function handleDemoSubmit(event) {
       event.preventDefault();
       
       const name = document.getElementById('name').value;
       const email = document.getElementById('email').value;
       const company = document.getElementById('company').value;

       // Log submission data (In a real app, this would send data to a CRM/backend)
       console.log('--- Demo Request Submitted ---');
       console.log('Name:', name);
       console.log('Email:', email);
       console.log('Company:', company);
       console.log('----------------------------');

       // Simulate form success feedback
       event.target.reset(); // Clear form fields
       modalMessage.textContent = `Thank you, ${name} from ${company}! We will contact you at ${email} shortly.`;
       modalMessage.classList.remove('hidden');

       // Close modal after a short delay
       setTimeout(closeModal, 4000);
   }

   // Close modal if user clicks outside of it
   demoModal.addEventListener('click', (e) => {
       if (e.target === demoModal) {
           closeModal();
       }
   });
</script>
